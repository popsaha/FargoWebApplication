

CREATE TABLE dbo.USER_MASTER
(
USER_ID BIGINT PRIMARY KEY IDENTITY(1,1),
USER_CODE VARCHAR(20),
FIRST_NAME VARCHAR(100),
LAST_NAME VARCHAR(100),
EMAIL_ID VARCHAR(100),
GENDER VARCHAR(10),
STREET VARCHAR(500),
LANDMARK VARCHAR(500),
CITY VARCHAR(100),
DISTRICT VARCHAR(100),
STATE_ID BIGINT,
COUNTRY_ID BIGINT,
PINCODE VARCHAR(10),
IMEI_NUMBER VARCHAR(100),
DATE_OF_BIRTH DATETIME,
CONTACT_NO VARCHAR(20),
ALTERNATE_CONTACT_NO VARCHAR(200),
PROFILE_PHOTO VARCHAR(1000),
PROFILE_PHOTO_URL VARCHAR(1000),
STORE_ID BIGINT,
ROLE_ID BIGINT,
USERNAME VARCHAR(100),
PASSWORD VARCHAR(100),
DATA_SOURCE VARCHAR(10),
IS_LOG_OUT BIT,
LAST_LOGGEDIN_ON DATETIME,
LAST_LOGGEDOUT_ON DATETIME,
IS_ACTIVE BIT,
CREATED_BY BIGINT,
CREATED_ON DATETIME,
MODIFIED_BY BIGINT,
MODIFIED_ON DATETIME,
DELETED_BY BIGINT,
DELETED_ON DATETIME
)

)
